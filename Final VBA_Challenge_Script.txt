worksheetname = ws.Name        'Headers        ws.Cells(1, 10).Value = "Ticker"        ws.Cells(1, 11).Value = "Yearly Difference"        ws.Cells(1, 12).Value = "Percentage Difference"        ws.Cells(1, 13).Value = "Total Volume"        ws.Cells(1, 16).Value = "Ticker"        ws.Cells(1, 17).Value = "Value"        ws.Cells(2, 15).Value = "Greatest % Increase"        ws.Cells(3, 15).Value = "Greatest % Decrease"        ws.Cells(4, 15).Value = "Greatest Total Volume"        'Dimensions    Dim x As Double    Dim OpeningValue As Double    Dim TotalVolume As Double    Dim GreatestPercentInc As Double    Dim GreatestPercentDec As Double    Dim GreatestTotalVol As Double    Dim GreatTotalVolTicker As String    Dim IncTicker As String    Dim DecTicker As String 2).End(xlUp).Row    'For Loop through a single worksheet for Ticker and Total Volume    For x = 2 To LastRow    'For Zero Values in Column "3"        If ws.Cells(x, 3).Value = 0 Then    Else        'Determining Stock Value Total and tickers    TotalVolume = TotalVolume + ws.Cells(x, 7).Value    'Opening Value     If ws.Cells(x, 2).Value = ws.Cells(2, 2) Then            OpeningValue = ws.Cells(x, 3).Value        End If        'If Ticker and Volume         If ws.Cells(x + 1, 1).Value <> ws.Cells(x, 1).Value Then                        ws.Cells(SummaryRow, 10).Value = ws.Cells(x, 1).Value            ws.Cells(SummaryRow, 13).Value = TotalVolume                'Greatest Total Volume           If GreatestTotalVol < TotalVolume Then            GreatestTotalVol = TotalVolume                GreatTotalVolTicker = ws.Cells(x, 1).Value                     End If        ' Yearly Change Syntax            ws.Cells(SummaryRow, 11).Value = (ws.Cells(x, 6).Value - OpeningValue)                ' Percent Change Syntax            ws.Cells(SummaryRow, 12).Value = (ws.Cells(x, 6).Value - OpeningValue) / OpeningValue            ws.Cells(SummaryRow, 12).Value = ws.Cells(SummaryRow, 12).Value * 1            ws.Cells(SummaryRow, 12).NumberFormat = "0.00%"                    'Greatest Percent Values         If GreatestPercentInc < ws.Cells(SummaryRow,          ws.Cells(SummaryRow, 12).Interior.ColorIndex = 4                            End If                    TotalVolume = 0            SummaryRow = SummaryRow + 1                        End If        End If        Next x        ws.Cells(4, 16).Value = GreatTotalVolTicker    ws.Cells(4, 17).Value = GreatestTotalVol    ws.Cells(2, 16).Value = IncTicker    ws.Cells(2, 17).Value =     Next wsEnd Sub
